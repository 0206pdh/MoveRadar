## Problem Statement

이사 수요는 아파트 거래가 활성화되는 시점보다 **1~2개월 먼저** 발생한다.
시세 + 유동인구 + 계약 데이터를 결합하면 "지금 집중해야 할 지역"과 "보내야 할 메시지"를
마케팅 담당자보다 빠르게 자동으로 파악할 수 있다.

기존 마케팅 자동화는 CRM 참여 신호 기반 — 이미 관심을 표현한 사람에게만 반응.
MoveRadar는 아직 이사를 인지하지 못한 사람을 1~2개월 전에 탐지한다.
**관심 신호가 아닌 구조적 시장 신호** — 이게 차별점이다.

### price_timeseries_train_view
```sql
CREATE OR REPLACE VIEW price_timeseries_train_view AS
SELECT
    YYYYMMDD::DATE                        AS TS,
    SGG || '_' || EMD                     AS REGION_KEY,
    SGG,                                  -- REGION_ALERTS에서 pop 조인용
    AVG(MEME_PRICE_PER_M2)               AS MEME_PRICE_PER_M2
FROM KOREA_REAL_ESTATE_APARTMENT_MARKET_INTELLIGENCE.HACKATHON_2026.REGION_APT_RICHGO_MARKET_PRICE_M_H
WHERE REGION_LEVEL = 'emd'
  AND SGG IN ('서초구', '영등포구', '중구')
  AND YYYYMMDD::DATE < '2024-01-01'       -- 훈련 기간: 2021~2023
GROUP BY 1, 2, 3;

CREATE OR REPLACE VIEW price_timeseries_detect_view AS
SELECT
    YYYYMMDD::DATE                        AS TS,
    SGG || '_' || EMD                     AS REGION_KEY,
    SGG,                                  -- REGION_ALERTS에서 pop 조인용
    AVG(MEME_PRICE_PER_M2)               AS MEME_PRICE_PER_M2
FROM KOREA_REAL_ESTATE_APARTMENT_MARKET_INTELLIGENCE.HACKATHON_2026.REGION_APT_RICHGO_MARKET_PRICE_M_H
WHERE REGION_LEVEL = 'emd'
  AND SGG IN ('서초구', '영등포구', '중구')
  AND YYYYMMDD::DATE >= '2024-01-01'      -- 탐지 대상: 2024~
GROUP BY 1, 2, 3;
```

### pop_timeseries_train_view / pop_timeseries_detect_view
```sql
-- REGION_POPULATION_MOVEMENT: REGION_LEVEL은 'sgg'와 'sd'만 존재 (emd 없음 — EDA 확인)
-- SGG 단위로 전입인구 이상 탐지 → REGION_KEY = SGG 이름
-- YYYYMMDD 이미 DATE 타입, JOIN 불필요

CREATE OR REPLACE VIEW pop_timeseries_train_view AS
SELECT
    YYYYMMDD                               AS TS,
    SGG                                    AS REGION_KEY,  -- 구 단위 (emd 없으므로)

WHERE MOVEMENT_TYPE = '전입'
  AND REGION_LEVEL = 'sgg'
  AND SGG IN ('서초구', '영등포구', '중구')
  AND YYYYMMDD < '2024-01-01'              -- 훈련 기간: 2021~2023
GROUP BY 1, 2;

CREATE OR REPLACE VIEW pop_timeseries_detect_view AS
SELECT
    YYYYMMDD                               AS TS,
    SGG                                    AS REGION_KEY,
    SUM(POPULATION)                        AS POPULATION
FROM KOREA_REAL_ESTATE_APARTMENT_MARKET_INTELLIGENCE.HACKATHON_2026.REGION_POPULATION_MOVEMENT
WHERE MOVEMENT_TYPE = '전입'
  AND REGION_LEVEL = 'sgg'
  AND SGG IN ('서초구', '영등포구', '중구')
  AND YYYYMMDD >= '2024-01-01'             -- 탐지 대상: 2024~
GROUP BY 1, 2;
```

### customer_profile_view (Issue 2 수정: PROVINCE_CODE 조인, 날짜 필터 추가)
```sql
CREATE OR REPLACE VIEW customer_profile_view AS
SELECT
    m.DISTRICT_CODE,
    m.PROVINCE_CODE,
    m.CITY_KOR_NAME || '_' || m.DISTRICT_KOR_NAME         AS REGION_KEY,
    -- 주요 연령대 (가장 인구 많은 그룹)
    CASE
        WHEN p.AGE_30S = GREATEST(p.AGE_20S, p.AGE_30S, p.AGE_40S, p.AGE_50S) THEN '30대 중심'
        WHEN p.AGE_40S = GREATEST(p.AGE_20S, p.AGE_30S, p.AGE_40S, p.AGE_50S) THEN '40대 중심'
        ELSE '20-50대 혼합'
    END                                                     AS DOMINANT_AGE_GROUP,
    -- 소득 구간: PROVINCE_CODE 기준 (ASSET_INCOME_INFO는 시도 단위만 제공)
    CASE
        WHEN a.RATE_INCOME_OVER_70M > 0.2 THEN '고소득 지역'
        WHEN a.RATE_INCOME_OVER_70M > 0.1 THEN '중고소득 지역'
        ELSE '중소득 지역'
    END                                                     AS INCOME_PROFILE
FROM SEOUL_DISTRICTLEVEL_DATA_FLOATING_POPULATION_CONSUMPTION_AND_ASSETS.GRANDATA.M_SCCO_MST m
LEFT JOIN KOREA_REAL_ESTATE_APARTMENT_MARKET_INTELLIGENCE.HACKATHON_2025Q2.REGION_MOIS_POPULATION_GENDER_AGE_M_H p
    ON m.DISTRICT_KOR_NAME = p.EMD
    AND p.REGION_LEVEL = 'emd'
    AND p.YYYYMMDD = '20250101'           -- 날짜 고정으로 중복 방지
LEFT JOIN SEOUL_DISTRICTLEVEL_DATA_FLOATING_POPULATION_CONSUMPTION_AND_ASSETS.GRANDATA.ASSET_INCOME_INFO a
    ON m.PROVINCE_CODE = a.PROVINCE_CODE  -- ASSET_INCOME_INFO는 시도 단위
    AND a.STANDARD_YEAR_MONTH = '202501'
WHERE m.CITY_KOR_NAME IN ('서초구', '영등포구', '중구')
  AND m.DISTRICT_CODE IS NOT NULL;        -- 법정동 없는 행 제외
```

### REGION_ALERTS
```sql
CREATE OR REPLACE TABLE REGION_ALERTS AS
SELECT
    p.REGION_KEY,
    m.CITY_KOR_NAME                        AS SGG,
    m.DISTRICT_KOR_NAME                    AS EMD,
    m.DISTRICT_CODE,
    m.DISTRICT_GEOM,                       -- pydeck GeoJson 폴리곤
    p.TS                                   AS ALERT_DATE,
    p.IS_ANOMALY                           AS PRICE_IS_ANOMALY,
    pop.IS_ANOMALY                         AS POP_IS_ANOMALY,
    p.PERCENTILE                           AS PRICE_SCORE,
    pop.PERCENTILE                         AS POP_SCORE,
    -- COMBINED_SCORE: 두 점수의 가중 평균 (시세 60%, 유동인구 40%)
    (COALESCE(p.PERCENTILE, 0) * 0.6 + COALESCE(pop.PERCENTILE, 0) * 0.4)
                                           AS COMBINED_SCORE,
    CASE
        WHEN p.IS_ANOMALY AND pop.IS_ANOMALY THEN '시세+전입인구 동시 경보'
        WHEN p.IS_ANOMALY THEN '시세 경보'
        WHEN pop.IS_ANOMALY THEN '전입인구 경보'
    END                                    AS ALERT_TYPE
FROM PRICE_ANOMALY_RESULTS p
LEFT JOIN POP_ANOMALY_RESULTS pop
    ON p.SGG = pop.REGION_KEY AND p.TS = pop.TS  -- pop은 SGG 단위, price는 EMD 단위
JOIN SEOUL_DISTRICTLEVEL_DATA_FLOATING_POPULATION_CONSUMPTION_AND_ASSETS.GRANDATA.M_SCCO_MST m
    ON (m.CITY_KOR_NAME || '_' || m.DISTRICT_KOR_NAME) = p.REGION_KEY
WHERE p.IS_ANOMALY OR pop.IS_ANOMALY;
```

### MARKETING_ALERTS (최종 출력 테이블, Issue 3 수정: TOP 15 + NULL 안전처리)
```sql
-- 컬럼: REGION_KEY, SGG, EMD, DISTRICT_CODE, DISTRICT_GEOM,
--       ALERT_DATE, ALERT_TYPE, COMBINED_SCORE,
--       DOMINANT_AGE_GROUP, INCOME_PROFILE, MARKETING_COPY
CREATE OR REPLACE TABLE MARKETING_ALERTS AS
SELECT
    r.REGION_KEY, r.SGG, r.EMD, r.DISTRICT_CODE, r.DISTRICT_GEOM,
    r.ALERT_DATE, r.ALERT_TYPE, r.COMBINED_SCORE,
    cp.DOMINANT_AGE_GROUP,
    cp.INCOME_PROFILE,
    SNOWFLAKE.CORTEX.COMPLETE(
        'mistral-large2',
        '이사 수요 급증 신호가 탐지된 지역에 맞는 인터넷/렌탈 마케팅 문구를 생성해줘.

지역: ' || COALESCE(r.SGG, '미상') || ' ' || COALESCE(r.EMD, '미상') || '
경보 유형: ' || COALESCE(r.ALERT_TYPE, '이상 탐지') || '
결합 이상 지수: ' || ROUND(r.COMBINED_SCORE, 2) || '
주요 고객층: ' || COALESCE(cp.DOMINANT_AGE_GROUP, '일반') || '
소득 수준: ' || COALESCE(cp.INCOME_PROFILE, '중소득 지역') || '

조건:
- 50자 이내 짧은 광고 문구
- 이사 불안 해소 + 서비스 편의성 강조
- 지역명 포함
- 자연스러운 한국어
- 인터넷 또는 렌탈 서비스 중 하나 선택해서 구체적으로'
    )                                      AS MARKETING_COPY
FROM (
    SELECT * FROM REGION_ALERTS
    WHERE COMBINED_SCORE > 0.80
    ORDER BY COMBINED_SCORE DESC
    LIMIT 15                               -- API 실패 리스크 최소화 + 데모에 충분
) r
JOIN customer_profile_view cp
    ON r.DISTRICT_CODE = cp.DISTRICT_CODE;
```

## Cortex Anomaly Detection 실행 패턴 (Issue 1 수정: train/detect 뷰 분리)

```sql
-- 1. 시세 모델 학습 (train_view = 2021~2023만 포함)
CREATE OR REPLACE SNOWFLAKE.ML.ANOMALY_DETECTION price_anomaly_model(
    INPUT_DATA        => SYSTEM$REFERENCE('VIEW', 'price_timeseries_train_view'),
    SERIES_COLNAME    => 'REGION_KEY',
    TIMESTAMP_COLNAME => 'TS',
    TARGET_COLNAME    => 'MEME_PRICE_PER_M2',
    LABEL_COLNAME     => NULL
);

-- 2. 시세 이상 탐지 (detect_view = 2024 이후만 포함)
CALL price_anomaly_model!DETECT_ANOMALIES(
    INPUT_DATA        => SYSTEM$REFERENCE('VIEW', 'price_timeseries_detect_view'),
    TIMESTAMP_COLNAME => 'TS',
    TARGET_COLNAME    => 'MEME_PRICE_PER_M2',
    OUTPUT_TABLE_NAME => 'PRICE_ANOMALY_RESULTS'
);

-- 3. 전입인구 모델 학습 (REGION_POPULATION_MOVEMENT 기반)
CREATE OR REPLACE SNOWFLAKE.ML.ANOMALY_DETECTION pop_anomaly_model(
    INPUT_DATA        => SYSTEM$REFERENCE('VIEW', 'pop_timeseries_train_view'),
    SERIES_COLNAME    => 'REGION_KEY',
    TIMESTAMP_COLNAME => 'TS',
    TARGET_COLNAME    => 'POPULATION',      -- REGION_POPULATION_MOVEMENT의 실제 컬럼명
    LABEL_COLNAME     => NULL
);

-- 4. 전입인구 이상 탐지
CALL pop_anomaly_model!DETECT_ANOMALIES(
    INPUT_DATA        => SYSTEM$REFERENCE('VIEW', 'pop_timeseries_detect_view'),
    TIMESTAMP_COLNAME => 'TS',
    TARGET_COLNAME    => 'POPULATION',
    OUTPUT_TABLE_NAME => 'POP_ANOMALY_RESULTS'
);
```

> **리스크 체크:** `SELECT REGION_KEY, COUNT(*) FROM price_timeseries_view GROUP BY 1`
> 법정동별 최소 24개 포인트(2년치) 확보 여부를 EDA 시 반드시 확인.
> 부족하면 EMD → SGG(구 단위)로 집계 레벨 올리는 fallback 사용.

## Streamlit 대시보드

```python
# streamlit_app.py (Snowflake Native App)
import streamlit as st
import pydeck as pdk
import json
import pandas as pd
from snowflake.snowpark.context import get_active_session

session = get_active_session()

st.title("MoveRadar — 이사 수요 경보 대시보드")

# 경보 데이터 로드
alerts = session.sql("""
    SELECT SGG, EMD, ALERT_DATE, ALERT_TYPE, COMBINED_SCORE,
           DOMINANT_AGE_GROUP, INCOME_PROFILE, MARKETING_COPY,
           ST_ASGEOJSON(DISTRICT_GEOM) AS GEOJSON  -- WKT → GeoJSON 변환
    FROM MARKETING_ALERTS
    ORDER BY COMBINED_SCORE DESC
""").to_pandas()

# GeoJSON 파싱
alerts["geometry"] = alerts["GEOJSON"].apply(json.loads)

# 경보 없을 때 빈 화면 처리 (Performance Issue 수정)
if alerts.empty:
    st.warning("현재 탐지된 경보가 없습니다. 파이프라인을 먼저 실행해 주세요.")
    st.stop()

# 지역 선택 — index 기반으로 한국어 문자열 비교 오류 방지 (Issue 7 수정)
st.subheader("경보 지역 선택")
selected_idx = st.selectbox(
    "지역 선택",
    options=range(len(alerts)),
    format_func=lambda i: f"{alerts.iloc[i]['SGG']} {alerts.iloc[i]['EMD']}"
)
row = alerts.iloc[selected_idx]

col1, col2 = st.columns([3, 2])

with col1:
    # pydeck 지도
    layer = pdk.Layer(
        "GeoJsonLayer",
        data=alerts.to_dict("records"),
        get_geometry="geometry",
        get_fill_color="[255 * COMBINED_SCORE, 100, 50, 160]",  # f-string 아님 (Issue 5)
        get_line_color=[80, 80, 80],
        lineWidthMinPixels=1,
        pickable=True,
    )
    st.pydeck_chart(pdk.Deck(
        layers=[layer],
        initial_view_state=pdk.ViewState(latitude=37.55, longitude=126.98, zoom=12),
        tooltip={"text": "{SGG} {EMD}\n경보: {ALERT_TYPE}\n점수: {COMBINED_SCORE}"}
    ))

with col2:
    st.subheader(f"{row['SGG']} {row['EMD']}")
    st.metric("결합 경보 점수", f"{row['COMBINED_SCORE']:.2f}")
    st.write(f"**경보 유형:** {row['ALERT_TYPE']}")
    st.write(f"**주요 고객층:** {row['DOMINANT_AGE_GROUP']} / {row['INCOME_PROFILE']}")
    st.divider()
    st.subheader("LLM 생성 마케팅 문구")
    st.info(row["MARKETING_COPY"])
```

> **DISTRICT_GEOM 처리 주의:** M_SCCO_MST의 DISTRICT_GEOM은 Geography 타입.
> Streamlit에서 쓰려면 `ST_ASGEOJSON(DISTRICT_GEOM)` 으로 GeoJSON 문자열 변환 필요.
